# Omega Unit Tests


#####################
# Omega Test Function
#####################

function(add_omega_test test_name exe_name source_files mpi_args)

  # Create the executable
  add_executable(${exe_name} ${source_files})

  # Link the library
  target_link_libraries(
    ${exe_name}
    PRIVATE
    ${OMEGA_LIB_NAME}
    OmegaLibFlags
  )

  # Add the test command
  if (mpi_args)
    add_test(
      NAME ${test_name}
      COMMAND ${MPI_EXEC} ${mpi_args} -- ./${exe_name}
    )

  else()
    add_test(
      NAME ${test_name}
      COMMAND ./${exe_name}
    )
  endif()

endfunction()
